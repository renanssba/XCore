goto *arg1

waypoint event_statement
say "" "\observedPerson escuta latidos em uma parte da trilha. Um cachorro grande avança, lenta e ameaçadoramente, na direção de um garoto. É \encounterPerson."
say_auto "\observedPerson se aproxima com cuidado, e tenta resolver a situação acalmando o animal."
end_script



waypoint help_success
say "" "\observedPerson acalma o cachorro, que só estava assustado. O menino agradece."
end_script

waypoint help_fail
say "" "O cachorro não vai muito com a cara de \observedPerson. No momento em que ela se aproxima, o cão rosna e avança na direção dela. Ela sai de fininho."
end_script
