goto *arg1

waypoint event_statement
say "" "O casal encontra um coelhinho ferido, precisando de ajuda!"

set_var guts_action_name "Encorajar"
set_var intelligence_action_name "Tratar"
set_var charisma_action_name "Tranquilizar"
end_script

waypoint guts_action
say "" "\active faz um discurso inspirador!"
say "" "O coelhinho se sente mais motivado!"
end_script

waypoint intelligence_action
say "" "\active improvisa um curativo!"
say "" "O coelhinho se sente um pouco melhor!"
end_script

waypoint charisma_action
say "" "\active tranquiliza o coelhinho com um gesto acolhedor!"
say "" "O pobre animal se emociona!"
end_script

waypoint charisma_success

end_script
