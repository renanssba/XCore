if #ap < 2
  goto_script "not_enough_ap"
  end_script
endif


set_var hide_buttons 1
update_ui


waypoint sim
say_auto "Você pode gastar energia trabalhando para ganhar dinheiro. Quanto AP você quer gastar?"
choices "Gastar 2 AP -> $60" ap2 "Gastar 3 AP -> $120" ap3 "Cancelar" end

waypoint ap2
set_var ap_spent 2
goto get_money

waypoint ap3
set_var ap_spent 3
goto get_money



waypoint get_money
spend_ap ap_spent
update_ui
fade_out 0.7
wait 0.3

play_sfx "ui_money"
if ap_spent == 2
  set_var gain 60
endif
if ap_spent == 3
  set_var gain 120
endif
add_var money gain
update_ui
say "Você trabalhou bastante e ganhou \vsn[gain] Dinheiro Celestial!"

fade_in 0.7
wait 0.5
goto end



waypoint end
set_var hide_buttons 0
update_ui
