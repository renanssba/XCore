say "" "PART TIME JOB"

if #ap < 2
  goto_script "not_enough_ap"
  end_script
endif


set_var hide_buttons 1
update_ui


waypoint sim
say_auto "part_time_job/say_0"
choices "part_time_job/choices_0_0" ap2 "part_time_job/choices_0_1" ap3 "choices/cancel" end

waypoint ap2
set_var ap_spent 2
goto get_money

waypoint ap3
set_var ap_spent 3
goto get_money



waypoint get_money
spend_ap ap_spent
update_ui
fade_out 0.7
wait 0.3

play_sfx "ui_money"
if ap_spent == 2
  set_var gain 60
endif
if ap_spent == 3
  set_var gain 120
endif
add_var money gain
update_ui
say "part_time_job/say_1"

fade_in 0.7
wait 0.5
goto end



waypoint end
set_var hide_buttons 0
update_ui
