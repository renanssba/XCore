if #ap < 2
  goto_script "not_enough_ap"
  end_script
endif


set_var hide_buttons 1
update_ui

say_auto "You can spend AP to get money. Do you want to work?"
choices "Yes" sim "No" nao


waypoint sim
say_auto "The more AP you spend, the more money you'll get. How many AP do you want to spend?"
choices "Spend 2 AP" ap2 "Spend 3 AP" ap3 "Spend 4 AP" ap4 "Spend 5 AP" ap5 "Cancelar" nao

waypoint ap2
set_var ap_spent 2
goto start_minigame

waypoint ap3
set_var ap_spent 3
goto start_minigame

waypoint ap4
set_var ap_spent 4
goto start_minigame

waypoint ap5
set_var ap_spent 5
goto start_minigame



waypoint start_minigame
spend_ap ap_spent
update_ui
fade_out 0.7
load_scene "Minigame - Ice Cream"
wait 0.5
goto end

goto end


waypoint end
