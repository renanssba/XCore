goto *arg1

waypoint event_statement
say "" "\girl buys a pot of jam, but seems to have a hard time opening it. \boy asks to help her."
end_script



waypoint guts_success
say "" "Using a little bit of strength, \boy easily opens the pot. The jam tastes really good."
end_script

waypoint intelligence_success
say "" "By reading the small letters, \boy locates the pot opening instructions. The jam tastes really good."
end_script

waypoint charisma_success
say "" "\boy goes back to the store and speaks with the jam seller, convincing him the pot has some kind of flaw."
say "The seller apologizes and gives them back the money, and some other (easier to open) jam pots!"
end_script



waypoint guts_fail
say "" "\boy tries to open the pot using some strength but fails miserably. Enraged, he applies too much force and wrecks the pot in half."
end_script

waypoint intelligence_fail
say "" "\boy tries to understand how to open the pot, but the complicated words in the package instructions only make him very confused and frustrated. He fails the task."
end_script

waypoint charisma_fail
say "" "goes back to the store and speaks with the jam seller, trying to convince him the pot has some kind of flaw."
say "The seller gets really angry and tells \boy to piss off."
end_script
