
fade_out 0.5
bg "shopping"
fade_in 0.5


say "Atendente" "Oi Fertiliel, seja bem-vinda à loja do Mundo dos Anjos!"

waypoint store_loop
say "Gostaria de comprar alguma coisa?"

show_shop true


if item_id != -1
  if item_price <= money
    say "Você quer comprar \item_name por \item_price?"
	choices "Sim" sim "Não" nao
  else
    say "Me desculpe, eu acho que você não tem dinheiro o suficiente para comprar este item."
  endif
else
  goto end
endif
goto store_loop



waypoint sim
subtract_var money item_price
update_ui
say "*Você adquiriu \item_name!*"
add_item item_id 1
goto store_loop


waypoint nao
goto store_loop


waypoint end
say "Obrigada, volte sempre!"


fade_out 0.5
bg null
update_ui
fade_in 0.5
